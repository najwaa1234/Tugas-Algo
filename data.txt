#include <iostream>
#include <fstream>

    char kelas[10];
    float ipk;
};

    char pilihan;
    cout << "\nKembali ke Menu Utama(y/t)? ";

    while(getchar() != '\n');
}

    char fileName[50];

    cout << "\nINPUT DATA\n";
    cout << "========================================\n";

        cout << "NIM   : ";

}

void searching() {

    ifstream fp;
    data d[100];
    char fileName[50];
    char cari[20];
    int i=0;

    cout << "\nSEARCHING DATA\n";
    cout << "========================================\n";
    cout << "Nama file : ";
    cin >> fileName;

    fp.open(fileName);

    if(!fp) {
        cout << "File tidak ditemukan!\n";
        tanyaKembali();
        return;
    }

    while(fp >> d[i].nim >> d[i].kelas >> d[i].ipk) {
        i++;
    }

    fp.close();

    int n = i;

    cout << "Masukkan NIM yang dicari : ";
    cin >> cari;

    strcpy(d[n].nim, cari); // sentinel

    i = 0;
    while(strcmp(d[i].nim, cari) != 0) {
        i++;
    }

    if(i < n) {
        cout << "\nData ditemukan\n";
        cout << "-----------------------------\n";
        cout << "NIM   : " << d[i].nim << endl;
        cout << "Kelas : " << d[i].kelas << endl;
        cout << "IPK   : " << d[i].ipk << endl;
    }
    else {
        cout << "\nData tidak ditemukan\n";
    }

    cout << "========================================\n";
    tanyaKembali();
}

void sorting() {

    ifstream fp;
    data d[100], temp;
    char fileName[50];
    int n=0;

    cout << "\nSORTING DATA (Shell Sort)\n";
    cout << "========================================\n";
    cout << "Nama file : ";
    cin >> fileName;

    fp.open(fileName);

    if(!fp) {
        cout << "File tidak ditemukan!\n";
        tanyaKembali();
        return;
    }

    while(fp >> d[n].nim >> d[n].kelas >> d[n].ipk) {
        n++;
    }

    fp.close();

    int gap,i,j;

    for(gap = n/2; gap > 0; gap /= 2) {
        for(i = gap; i < n; i++) {

            temp = d[i];

            for(j = i; j >= gap && d[j-gap].ipk > temp.ipk; j -= gap) {
                d[j] = d[j-gap];
            }

            d[j] = temp;
        }
    }

    cout << "\n========================================\n";
    cout << "NIM\t\tKELAS\t\tIPK\n";
    cout << "========================================\n";

    for(i=0;i<n;i++) {
        cout << d[i].nim << "\t" << d[i].kelas << "\t\t" << d[i].ipk << endl;
    }

    cout << "========================================\n";

    tanyaKembali();
}

}
